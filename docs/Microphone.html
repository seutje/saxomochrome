Microphone → Tone.ExternalInput

Opens up the default source (typically the microphone).

new Tone.Microphone ()
code
//mic will feedback if played through master
 var mic = new Tone.Microphone();
 mic.open(function(){
 	//start the mic at ten seconds
 	mic.start(10);
 });
 //stop the mic
 mic.stop(20);
